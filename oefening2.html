\[ \int \sqrt{2} \cdot x^{-3/2} \, dx = -2\sqrt{2} \cdot x^{-1/2} \] \[ \Rightarrow -\frac{2\sqrt{2}}{\sqrt{x}} = \frac{2\sqrt{2x}}{x} \]
